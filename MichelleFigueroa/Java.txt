IterativeDemo

package com.amdocs;

public class IterativeDemo {

	public static void main(String[] args) {
		
		for(int i=1; i <= 10; i++)
		{
		System.out.println("Hello Java Developers");
		}
		System.out.println("Outside Loop");
		
		int i=1;
		while(i <= 7)
		{
			System.out.println("Hello Java Developers");
			i++;
		}
		System.out.println("Outside Loop");
		
		int h=1;
		do
		{
			System.out.println("Hello Java Developers");
			h++;
		}
		while(h <= 7);
		System.out.println("Outside Loop");
		
		for(int row=1; row <= 5; row++) // Outer Loop
		{
			for(int col=1; col <= 5; col++) // Inner Loop
			{
				System.out.print(row+" ");
			}
			
			System.out.println();
		}
		
		System.out.println("Outside Loop");

	}

}

BranchingDemo

package com.amdocs;

public class BranchingDemo {

	public static void main(String[] args) {
		
		for(int i=1; i <= 10; i++)
		{
			if(i == 5)
			{
				break; // terminate your loop
			}
			else
			{
				System.out.println(i+". Hello Java Developers");
			}
		}
		System.out.println("Outside Loop");
		
		int h=1;
		for(; ; h++)
		{
			if(h == 5)
			{
				break; // terminate your loop
			}
			else
			{
				System.out.println(h+". Hello Java Developers");
			}
		}
		System.out.println("Outside Loop");
		
		for(int i=1; i <= 10; i++)
		{
			if(i == 5)
			{
				continue; // skip a turn in the loop
			}
			else
			{
				System.out.println(i+". Hello Java Developers");
			}
		}
		System.out.println("Outside Loop");
	}	
}
